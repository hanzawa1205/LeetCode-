### 翻转链表：
```
#三指针方法
def Reverse(head):
    if not head.next:
      return head
    pre = None
    while head:
        p = head.next
        head.next = pre
        pre = head
        head = p
    return pre
```

### 二分查找：
```
#排序数组
def find(nums,target):
    if len(nums) == 0:
       return False
    left = 0
    right = len(nums) - 1
    while left <= right:
        mid = (left + right) // 2
        if nums[mid] == target:
            return True
        elif nums[mid] > target:
            right -= 1
        else:
            left += 1
     return False
  ```
