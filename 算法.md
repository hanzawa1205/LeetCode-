### 翻转链表：
```
#三指针方法
def Reverse(head):
    if not head.next:
      return head
    pre = None
    while head:
        p = head.next
        head.next = pre
        pre = head
        head = p
    return pre
```

### 二分查找：
```
#排序数组
def find(nums,target):
    if len(nums) == 0:
       return False
    left = 0
    right = len(nums) - 1
    while left <= right:
        mid = (left + right) // 2
        if nums[mid] == target:
            return True
        elif nums[mid] > target:
            right -= 1
        else:
            left += 1
     return False

 ### 蛇形输出二维数组
> 只适用于N*N阶矩阵
```
def snake_print(nums):
    if len(nums) <= 1:
      return nums
    row = len(nums) - 1
    col = len(nums[0]) - 1
    r = 0
    c = 0
    res = []
    while r <= col and c <= row:
        if c % 2 == 0:            #列为奇数时从上至下，为偶数时从下至上
            res.append(nums[r][c])
            r += 1
            if r == row:
                res.append(nums[r][c])
                c += 1
        else:
            res.append(nums[r][c])
            r -= 1
            if r == 0:
                res.append(nums[r][c])
                c += 1
    return res
```

### 冒泡排序

def bubble_sort(nums):
    for i in range(len(nums)):
        for j in range(1,len(nums)-i):
            if nums[j] < nums[j-1]:
                nums[j], nums[j-1] = nums[j-1], nums[j]
```

    if l > r:         #不然会超过最大递归深度
        return
    i = l
    j = r

    while i < j:

            j -= 1
        if i < j:

            i += 1

def merge(l, r):
    res = []
    i = j =0
    while i < len(l) and j < len(r):
        if l[i] < r[j]:
            res.append(l[i])
            i += 1
        else:
            res.append(r[j])
            j += 1
    if i == len(l):
        for k in r[j:]:
            res.append(k)
    else:
        for k in l[i:]:
            res.append(k)
    return res


def merge_sort(nums):
    if len(nums) <= 1:
        return nums
    mid = len(nums) // 2
    left = merge_sort(nums[:mid])
    right = merge_sort(nums[mid:])
    return merge(left, right)
